protected override void OnModelCreating(ModelBuilder modelBuilder)  
{  
    if (Database.IsMySql())  
    {  
	    modelBuilder.Entity<BookAuthor>()  
        .HasKey(x => new { x.BookId, x.AuthorId });  
    modelBuilder.ApplyConfiguration(new BookConfig());           
    }  
}
